The component that represents shellbar actions. It is a container wrapper for all product actions and links (required element).

<fd-shellbar-actions [user]="user"
                     [userMenu]="userMenu"
                     [productSwitcher]="productSwitcher">
       <button fd-button fdType="standard">Custom Button</button>

       <fd-shellbar-action *ngFor="let action of actions"
                           [glyph]="action.glyph"
                           [callback]="action.callback"
                           [label]="action.label"
                           [notificationCount]="action.notificationCount"
                           [notificationLabel]="action.notificationLabel">
       </fd-shellbar-action>
</fd-shellbar-actions>
